Docker for mac

See the official instruction page for installation detail.

quick start

  • start docker from /Applications or launchpad or spotlight search Docker

  • docker basics, see examples

    $ docker run ubuntu /bin/echo 'Hello world'  # start a container
$ docker run -ti ubuntu /bin/bash            # start an interactive container

$ docker ps                 # list running containers
$ docker ps -a              # list all containers
$ docker logs <container>   # Shows us the standard output of a container.
$ docker stop <container>   # Stops running containers.
$ docker start <container>  # restart a containers.
$ docker rm <container>     # remover a containers.
$ docker images             # list all images
$ docker search ubuntu      # search for images
$ docker pull ubuntu        # download an image
$ docker pull ubuntu:14.04  # download an image with tag
$ docker push ouruser/sinatra

$ docker tag 5db5f8471261 ouruser/sinatra:devel
$ docker images ouruser/sinatra
$ docker rmi training/sinatra

$ docker run -d -P --name web training/webapp python app.py
$ docker inspect web
$ docker stop web
$ docker rm web

$ docker version
$ docker --help
$ docker attach --help

  • build a new image

    # method 1: modify and commit
$ docker run -t -i training/sinatra /bin/bash
root@0b2616b0e5a8:/# gem install json
root@0b2616b0e5a8:/# exit
$ docker commit -m "Added json gem" -a "Kate Smith" \
0b2616b0e5a8 ouruser/sinatra:v2
$ docker images
#
# method 2: build from a Dockerfile
$ mkdir sinatra
$ cd sinatra
$ touch Dockerfile
# save the following to the Dockerfile
    # This is a comment
    FROM ubuntu:14.04
    MAINTAINER Kate Smith <ksmith@example.com>
    RUN apt-get update && apt-get install -y ruby ruby-dev
    RUN gem install sinatra
$ docker build -t ouruser/sinatra:v2 .

  • mount host directory to a container

    $ docker run -d -P --name web -v /host/path:/container/path training/webapp python app.py
$ docker run --rm -it -v ~/.bash_history:/root/.bash_history ubuntu /bin/bash
$
# shared volume
$ docker create -v /dbdata --name dbstore training/postgres /bin/true
$ docker run -d --volumes-from dbstore --name db1 training/postgres
$ docker run -d --volumes-from dbstore --name db2 training/postgres
$ docker run --rm -v /foo -v awesome:/bar busybox top # remove volume

  • open gui apps in a docker container

    # install socat and xquartz first
$ brew install socat
$ socat TCP-LISTEN:6000,reuseaddr,fork UNIX-CLIENT:\"$DISPLAY\"
# in another window
$ open -a XQuartz
$ open -a /Applications/Docker/Docker\ Quickstart\ Terminal.app
# in the Docker CPL terminal
$ ifconfig # find ip address
$ docker run -ti -e DISPLAY=192.168.1.115:0 ubuntu /bin/bash
# should be able to apps gui
